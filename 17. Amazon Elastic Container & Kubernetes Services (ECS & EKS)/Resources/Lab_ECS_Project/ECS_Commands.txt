## Install Docker
sudo dnf update -y 
sudo dnf install docker -y 
sudo systemctl enable docker 
sudo systemctl start docker 
sudo usermod -aG docker ec2-user 
newgrp docker

Verify Docker Installation by running - docker --version


## Install PHP & Composer
sudo dnf install php-cli unzip git -y
php -r"copy('https://getcomposer.org/installer', 'composer-setup.php');" 
php composer-setup.php
sudo mv composer.phar /usr/local/bin/composer


## Install Dependencies (AWS SDK)
sudo dnf install php-simplexml -y
composer install

## Build Docker Image
docker build -t CloudFreeks-S3-PHP .

## Test Image using Variables

docker run -d -p 8080:80 \
 -e S3_BUCKET=<S3_BUCKET_NAME> \
 -e AWS_REGION=<AWS_REGION> \
 -e AWS_ACCESS_KEY_ID=<AWS_ACCESS_KEY_ID> \
 -e AWS_SECRET_ACCESS_KEY=<AWS_SECRET_ACCESS_KEY> \
 --name php-test cloudfreeks-s3-php
